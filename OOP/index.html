Activity 3

Simple Wallet System

Defining and calling methods to promote code reusability and organization.

import java.util.Scanner; public class RegalaSimpleWallet { public static void main(String[] args) { Scanner input = new Scanner(System.in); double balance = 3000.0; int successfulTransactions = 0; while (true) { System.out.println("\n======================="); System.out.println("===== WALLET MENU ====="); System.out.println("======================="); System.out.println("1. View Balance"); System.out.println("2. Add Money"); System.out.println("3. Spend Money"); System.out.println("4. Exit"); System.out.print("Choice: "); int choice = input.nextInt(); switch (choice) { case 1: System.out.println("Current Balance: " + balance); successfulTransactions++; break; case 2: System.out.print("Enter amount to add: "); double addAmount = input.nextDouble(); balance += addAmount; System.out.println("Successfully added " + addAmount); System.out.println("New Balance: " + balance); successfulTransactions++; break; case 3: System.out.print("Enter amount to spend: "); double spendAmount = input.nextDouble(); if (spendAmount > 1000) { System.out.println("Exceeds maximum spend"); } else if (spendAmount > balance) { System.out.println("Transaction Failed: Insufficient balance."); } else { balance -= spendAmount; System.out.println("Successfully spent " + spendAmount); System.out.println("New Balance: " + balance); successfulTransactions++; } break; case 4: System.out.println("Final Transaction Count: " + successfulTransactions); System.out.println("Exiting..."); input.close(); System.exit(0); break; default: System.out.println("Invalid choice. Please try again."); } } } }

public class RegalaGradeEvaluation { public static void main(String[] args) { double prelim = 85; double midterm = 78; double finals = 90; double average = (prelim + midterm + finals) / 3; String remark = ""; if (average >= 90 && average <= 100) { remark = "Excellent"; } else if (average >= 80 && average < 90) { remark = "Good"; } else if (average >= 75 && average < 80) { remark = "Fair"; } else { remark = "Failed"; } System.out.println("Prelim Grade: " + prelim); System.out.println("Midterm Grade: " + midterm); System.out.println("Final Grade: " + finals); System.out.printf("Average: %.2f\n", average); System.out.println("Remark: " + remark); } }

Activity 5

Personal Expennse Tracker

A void method that displays the program title. A non-void method that calculates the total expenses using three values: food, transportation, and other expenses. A non-void method that checks if the total expense exceeds a budget and returns a message. A void method that displays the total expense and the budget status.

import java.util.Scanner; public class ExpenseTrackerRegala { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); displayTitle(); System.out.print("Enter your monthly budget: "); double budget = scanner.nextDouble(); System.out.print("Enter food expenses: "); double food = scanner.nextDouble(); System.out.print("Enter transportation expenses: "); double transpo = scanner.nextDouble(); System.out.print("Enter other expenses: "); double otherExp = scanner.nextDouble(); double totalExp = calculateTot(food, transpo, otherExp); String status = checkBudget(totalExp, budget); double remaining = budget - totalExp; displayFinalResult(totalExp, status, remaining); scanner.close(); } public static void displayTitle() { System.out.println("======================================="); System.out.println(" PERSONAL EXPENSE TRACKER "); System.out.println("======================================="); } public static double calculateTot(double food, double trans, double misc) { return food + trans + misc; } public static String checkBudget(double total, double budget) { if (total > budget) { return "Warning: You have exceeded your budget!"; } else { return "Good job! You are within your budget."; } } public static void displayFinalResult(double total, String status, double remaining) { System.out.println("\n EXPENSE SUMMARY "); System.out.printf("Total Expenses: Php %.2f\n", total); System.out.printf("Remaining Budget: Php %.2f\n", remaining); System.out.println("Status: " + status); } }
